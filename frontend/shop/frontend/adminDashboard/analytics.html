async function fetchAnalyticsData() { const token = getAuthToken(); if (!token) { showToast('Session expired'); throw new Error('No auth token'); } const headers = getHeaders(); // SIMPLE API CALLS (no fallback) const kpis = await fetch(`${API_BASE_URL}/admin/analytics/kpis`, { headers }).then(res => res.json()); const revenueData = await fetch(`${API_BASE_URL}/admin/analytics/revenue`, { headers }).then(res => res.json()); const categoryData = await fetch(`${API_BASE_URL}/admin/analytics/categories`, { headers }).then(res => res.json()); const dailyOrders = await fetch(`${API_BASE_URL}/admin/analytics/daily-orders`, { headers }).then(res => res.json()); const insights = await fetch(`${API_BASE_URL}/admin/analytics/insights`, { headers }).then(res => res.json()); const topProducts = await fetch(`${API_BASE_URL}/admin/analytics/top-products`, { headers }).then(res => res.json()); const topCustomers = await fetch(`${API_BASE_URL}/admin/analytics/top-customers`, { headers }).then(res => res.json()); return { kpis, revenueData, categoryData, dailyOrders, insights, topProducts, topCustomers }; }
